1.3. Preprocessing¶

Define STFT Function

In [8]:
def stft(signal, Fs, frame_size, hop):
    '''
    <Input>
    Signal : signal to be interpreted
    Fs : sampling rate
    frame_size : window size (metric is second)
    hop : stride size (metric is second)
    
    <Output>
    X : matrix whose shape is (frequency x time)
    '''
    
    frame_sample = int(frame_size * Fs)
    hop_sample = int(hop * Fs)
    
    W = scipy.hanning(frame_sample)
    X = scipy.array([scipy.fft(W*signal[i:i+frame_sample])
                   for i in range(0, len(signal) - frame_sample, hop_sample)])
    return X.T

2.3. Network¶

  • Define CNN model network
In [13]:
def net(x, weights, biases):
    
    ## First convolution layer
    conv1 = tf.nn.conv2d(x, weights['conv1'], 
                         strides= [1, 1, 1, 1], 
                         padding = 'SAME')
    conv1 = tf.nn.relu(tf.add(conv1, biases['conv1']))
    
    ## Second convolution layer
    conv2 = tf.nn.conv2d(conv1, weights['conv2'], 
                         strides= [1, 1, 1, 1], 
                         padding = 'SAME')
    conv2 = tf.nn.relu(tf.add(conv2, biases['conv2']))
    
    maxp = tf.nn.max_pool(conv2, 
                          ksize = [1, p2_h, p2_w, 1], 
                          strides = [1, p2_h, p2_w, 1], 
                          padding = 'VALID')

    maxp_re = tf.reshape(maxp, [-1, conv_result_size])
    
    ### Fully connected
    hidden1 = tf.add(tf.matmul(maxp_re, weights['hidden1']), biases['hidden1'])
    hidden1 = tf.nn.relu(hidden1)
    output = tf.add(tf.matmul(hidden1, weights['output']), biases['output'])
    return {
        'conv1' : conv1,
        'conv2' : conv2,
        'maxp' : maxp,
        'output' : output}
